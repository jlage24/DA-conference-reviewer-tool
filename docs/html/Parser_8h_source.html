55inline string trim(const string& s) {
56 string result = s;
57 result.erase(remove(result.begin(), result.end(), '\r'), result.end());
58 size_t start = result.find_first_not_of(" \t");
59 if (start == string::npos) return "";
60 size_t end = result.find_last_not_of(" \t");
61 return result.substr(start, end - start + 1);

92 if (line == "#Parameters") { currentSection = PARAMETERS; continue; }
93 if (line == "#Control") { currentSection = CONTROL; continue; }
94 size_t hashPos = line.find('#');

98 line = trim(line);
99 if (line.empty()) continue;
100 stringstream ss(line);
101 string token;
102 try {
103 switch (currentSection) {
104 case SUBMISSIONS: {
105 Submission s;
106 getline(ss, token, ','); s.id = stoi(trim(token));
107 getline(ss, token, ',');
108 token.erase(remove(token.begin(), token.end(), '"'), token.end());
109 s.title = trim(token);
110 getline(ss, token, ','); s.authors = trim(token);
111 getline(ss, token, ','); s.email = trim(token);
112 getline(ss, token, ','); s.primary = stoi(trim(token));
113 getline(ss, token, ','); token = trim(token);
114 s.secondary = token.empty() ? 0 : stoi(token);
115 submissions.push_back(s);
116 break;
117 }
118 case REVIEWERS: {
119 Reviewer r;
120 getline(ss, token, ','); r.id = stoi(trim(token));
121 getline(ss, token, ','); r.name = trim(token);
122 getline(ss, token, ','); r.email = trim(token);
123 getline(ss, token, ','); r.primary = stoi(trim(token));

132 getline(ss, value); value = trim(value);
133 if (key == "MinReviewsPerSubmission") params.minReviewsPerSubmission = stoi(value);

135 else if (key == "PrimaryReviewerExpertise") params.primaryReviewerExpertise = stoi(value);

158 submissions.clear();
159 reviewers.clear();
